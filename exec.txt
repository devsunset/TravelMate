# TripMate 로컬 실행 명령어
# 백엔드와 프론트(웹)를 각각 터미널에서 실행한 뒤, Firefox에서 http://localhost:8080 으로 접속하면 됩니다.

# ----------------------------------------------------------------------
# 1. 백엔드 (Node.js API 서버) — 로컬 실행
# ----------------------------------------------------------------------
# 터미널 1에서 실행. API는 http://localhost:3000 에서 동작합니다.
# 최초 1회: .env 설정 후 npm install 필요

cd travel_mate_backend
npm install
node src/app.js

# (또는 프로젝트 루트에서)
# cd travel_mate_backend && npm install && node src/app.js


# ----------------------------------------------------------------------
# 2. 프론트 (Flutter 웹) — Firefox에서 확인
# ----------------------------------------------------------------------
# 터미널 2에서 실행. 브라우저에서 http://localhost:8080 접속 (Firefox 권장)

cd travel_mate_app
flutter pub get
flutter run -d web-server --web-port=8080 --web-hostname=0.0.0.0

# 실행 후 Firefox 주소창에 입력: http://localhost:8080


# ----------------------------------------------------------------------
# 참고
# ----------------------------------------------------------------------
# - 백엔드 .env: travel_mate_backend/.env (DB, FIREBASE_SERVICE_ACCOUNT_KEY_PATH 등)
# - 앱 API 주소: travel_mate_app/lib/app/constants.dart 의 apiBaseUrl (기본 http://localhost:3000)

# TripMate 로컬 실행
# 터미널 2개: 1) 백엔드  2) 프론트 웹 → Firefox에서 http://localhost:8080

# ---------- 1. 백엔드 (API http://localhost:3000) ----------
cd travel_mate_backend
npm install
node src/app.js

# ---------- 2. 프론트 웹 (Firefox 접속 http://localhost:8080) ----------
cd travel_mate_app
flutter pub get
flutter run -d web-server --web-port=8080 --web-hostname=0.0.0.0
